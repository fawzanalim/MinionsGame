        function newGame(app)
            cla(app.UIAxes);
            hold(app.UIAxes, 'on');
            
            app.PlayerName = "FAW";
            app.Lives = 5;
            app.Score = 0;
            app.AngleSet = 0;
            app.SpeedSet = 0;
            app.ShootingAngle.Value = app.ShootingAngle.Limits(1);
            app.Slider.Value = app.Slider.Limits(1);
            app.ScoreValueGame.Text = "0";
            updateNextScore(app);
            app.img_lives1.Visible = 'on';
            app.img_lives2.Visible = 'on';
            app.img_lives3.Visible = 'on';
            app.img_lives4.Visible = 'on';
            app.img_lives5.Visible = 'on';
            
        end

        function mainGame(app)

            try
                while (app.Lives > 0)                       % 1 loop represents 1 shot
                    app.snowball.Position = [8 8 25 25];    % Sets the position of the projectile in the window
                    app.Va = setWindVelocity(app);          % Sets wind velocity
                    app.AngleSet = 0;                       % 0 => Angle is not set yet
                    app.SpeedSet = 0;                       % 0 => Speed is not set yet
                    setTarget(app);                         % Sets the position of the target
                     
                    setAngle(app);                          % Moves the Angle gauge up and down until Play button is clicked
                    setSpeed(app);                          % Moves the Speed slider up and down until Play button is clicked
                    calculate(app);                         % Calculates the x and y coordinates of the projectile
                    shoot(app);                             % Shoots the projectile
                    scoring(app);                           % Calculates the score after each shot
                    updateNextScore(app);                   % Updates Next High Score & Current Position in the Game window
                    pause(1);
                end
            
            
            app.TabGroup.SelectedTab = app.GameOverTab;
            app.ScoreValueScore.Text = num2str(app.Score);  % Update the score in the game over window

            catch
            end
        end
        
        function readAllHighScores(app)
            try
                file = fopen(app.Filename);
                data = textscan(file,'%s');
                fclose(file);
                data = string(data{:});                                             % Reads the whole file into one string
                app.HighScorers = transpose(reshape(data(1:18), 6, 3));             % first 18 elements converted to a 6x3 matrix
                app.HighScores = transpose(str2double(reshape(data(19:36), 6, 3))); % last 18 elements converted to another 6x3 matrix
            catch 
                %If file doesnt exist, or something went wrong, set default high scores
                app.HighScorers = [ "EAA" "EBB" "ECC" "EDD" "EEE" "EFF";
                                    "NAA" "NBB" "NCC" "NDD" "NEE" "NFF"; 
                                    "HAA" "HBB" "HCC" "HDD" "HEE" "HFF"];
                app.HighScores = [49 39 29 19 9 3; 48 38 28 18 8 1; 47 37 27 17 7 2];
            end
        end
        
        
        function Va = setWindVelocity(app)
            if (app.Difficulty ~= 3)        % wind velocity = 0 for easy and normal mode
                    Va = 0;
            else
                Va_up = 0.25;   % max wind velocity
                Va_low = 0.1;   % min wind velocity

                % set a random value between 0.1 and 0.25
                Va = rand();
                if (Va > Va_up)
                    Va = mod(Va, Va_up);
                end
                if (Va < Va_low)
                    Va = Va + Va_low;
                end
            end

            app.WindValue.Text = num2str(round(Va, 2));

            
        end

        function setTarget(app)
            % Graph is 8 unit tall
            % Target need to between 1 and 7 with 1 unit to spare from top and bottom
            % Position on the x-axis will be fixed, 19.75
            if (app.Difficulty == 1)
                app.GoalRange = [4 6];      % Target is fixed in the middle for easy mode
            else
                app.GoalRange(1) = round(1 + (7-1) .* rand(1)); % For normal and hard, low point of target is random between 1 and 6
                app.GoalRange(2) = app.GoalRange(1) + 2;        % high point of target is 2 unit above low point
            end
            try
                delete(app.GoalBars);   %B efore changing new position of the target, try to delete old one if exists
            catch
            end
            app.vector.Position = [1720 20+95*(app.GoalRange(1)) 145 194];  % Position of the image target
            %Position of the target in the graph
            app.GoalBars(1) = rectangle(app.UIAxes, 'Position',[19.75,app.GoalRange(2),0.75,10-app.GoalRange(2)],'FaceColor',[0 .5 .5]);
            app.GoalBars(2) = rectangle(app.UIAxes, 'Position',[19.75,0,0.75,app.GoalRange(1)],'FaceColor',[0 .5 .5]);
        end
        
        function setHighScores(app)
            % Set High Scorers Name
            app.FirstName.Text = app.HighScorers(app.HighScoreMode, 1);
            app.SecondName.Text = app.HighScorers(app.HighScoreMode, 2);
            app.ThirdName.Text = app.HighScorers(app.HighScoreMode, 3);
            app.FourthName.Text = app.HighScorers(app.HighScoreMode, 4);
            app.FifthName.Text = app.HighScorers(app.HighScoreMode, 5);
            app.SixthName.Text = app.HighScorers(app.HighScoreMode, 6);
            
            % Set High Scores
            app.FirstScore.Text = num2str(app.HighScores(app.HighScoreMode, 1));
            app.SecondScore.Text = num2str(app.HighScores(app.HighScoreMode, 2));
            app.ThirdScore.Text = num2str(app.HighScores(app.HighScoreMode, 3));
            app.FourthScore.Text = num2str(app.HighScores(app.HighScoreMode, 4));
            app.FifthScore.Text = num2str(app.HighScores(app.HighScoreMode, 5));
            app.SixthScore.Text = num2str(app.HighScores(app.HighScoreMode, 6));

            % Disable the button for selected mode
            if (app.HighScoreMode == 1)
                set(app.ScoreBtnEasy,'Enable','off');
                set(app.ScoreBtnNormal,'Enable','on');
                set(app.ScoreBtnHard,'Enable','on');
            elseif (app.HighScoreMode == 2)
                set(app.ScoreBtnEasy,'Enable','on');
                set(app.ScoreBtnNormal,'Enable','off');
                set(app.ScoreBtnHard,'Enable','on');
            else
                set(app.ScoreBtnEasy,'Enable','on');
                set(app.ScoreBtnNormal,'Enable','on');
                set(app.ScoreBtnHard,'Enable','off');
            end
        end
        
        
        
        
        function setAngle(app)
            app.Instructions.Text = "Click on Play button to set the angle";
            % When up = 1, go from 0 to 90, 
            % When up = 0, go from 90 to 0
            up = 1; 
            while (app.AngleSet == 0)
                if (up == 1 && app.ShootingAngle.Value == 89)
                    up = 0;
                end
    
                if (up == 0 && app.ShootingAngle.Value == 1)
                    up = 1;
                end
    
                if (up == 1)
                    app.ShootingAngle.Value = app.ShootingAngle.Value+1;
                else
                    app.ShootingAngle.Value = app.ShootingAngle.Value-1;
                end
                pause(0.01);    % Pause to nicely animate it
            end
        end
        
        function setSpeed(app)
            app.Instructions.Text = "Click on Play button to set the speed";
            % When up = 1, go from low to high, 
            % When up = 0, go from high to low
            up = 1;
            while (app.SpeedSet == 0)
                if (up == 1 && app.Slider.Value == app.Slider.Limits(2))
                    up = 0;
                end
    
                if (up == 0 && app.Slider.Value == app.Slider.Limits(1))
                    up = 1;
                end

                if (up == 1 && app.Slider.Value+0.125 <= app.Slider.Limits(2))
                    app.Slider.Value = app.Slider.Value+0.125;
                elseif (up == 0 && app.Slider.Value-0.125 >= app.Slider.Limits(1))
                    app.Slider.Value = app.Slider.Value-0.125;
                end
                pause(0.01);    % Pause to nicely animate it
            end
        end

        function calculate(app)
            
            u = app.Slider.Value;           % Get speed from speed slider
            theta = app.ShootingAngle.Value;% Get angle from angle gauge
            
            app.Position_x = linspace(0, 20, 100);  % Get 100 points from 0 to 20
            % Calculate y-coords using projectile motion formula, considering wind velocity
            app.Position_y = ((u.*app.Position_x.*sind(theta))./(u.*cosd(theta) + app.Va)) - ((0.5).*app.g.*(app.Position_x.^2))./((u.*cosd(theta)+app.Va).^2);
        end

        function shoot(app)
            app.Instructions.Text = "";             % Hide instruction message during shooting animation
            set(app.PlayButtonGame,'Enable','off'); % Disable Play button during shooting
            
            % Animate the projectile
            for i = 1:length(app.Position_x)
                plot(app.UIAxes, app.Position_x(i), app.Position_y(i), 'c*');
                app.snowball.Position = [(8+app.Position_x(i)*1793/20) (8+app.Position_y(i)*(952)/10) 25 25];
                pause(0.005)
            end
            
            set(app.PlayButtonGame,'Enable','on')   % Enable Play button after projectile was shot
        end

        function scoring(app)
            % If the y coordinate in between the target range
            % When x coordinate is 20, it considered as target hit
            % Give a small room for error margin, 0.1 is good enough iguess
            if (app.Position_y(100)+0.1 > app.GoalRange(1) && app.Position_y(100)-0.1 < app.GoalRange(2))
                app.Score = app.Score + 10;
                app.ScoreValueGame.Text = num2str(app.Score);
            else
                % If target is missed, remove one live at a time by hiding the images
                app.Lives = app.Lives - 1;
                if(app.Lives == 4)
                    app.img_lives5.Visible = 'off';
                elseif(app.Lives == 3)
                    app.img_lives4.Visible = 'off';
                elseif(app.Lives == 2)
                    app.img_lives3.Visible = 'off';
                elseif(app.Lives == 1)
                    app.img_lives2.Visible = 'off';
                else
                    app.img_lives2.Visible = 'off';
                end
            end
            
        end
        
        function updateNextScore(app)
            % Find the next high score index
            row = app.Difficulty;
            index = find(app.Score > app.HighScores(row, 1:6), 1); 
            
            % Update the next score in the game
            % If the Score is the highest, show it.
            if(app.Score == 0 || isempty(index))
                app.NextScoreValue.Text = num2str(app.HighScores(row, 6));
            elseif(index == 1)
                app.NextScoreValue.Text = num2str(app.Score);
            else
                app.NextScoreValue.Text = num2str(app.HighScores(row, index-1));

            end
            
            % Update the current position in the game window
            if(index == 1)
                app.CurrentPosValue.Text = "1st";
            elseif(index == 2)
                app.CurrentPosValue.Text = "2nd";
            elseif(index == 3)
                app.CurrentPosValue.Text = "3rd";
            elseif(index == 4)
                app.CurrentPosValue.Text = "4th";
            elseif(index == 5)
                app.CurrentPosValue.Text = "5th";
            elseif(index == 6)
                app.CurrentPosValue.Text = "6th";
            else
                app.CurrentPosValue.Text = "7th";
            end
        end
        
        function updateHighScores(app)
            
            row = app.Difficulty;
            index = find(app.Score > app.HighScores(row, 1:6), 1); %In case of same score, old score will be at the top.

            if(index)
                app.HighScores(row, 1:6) = [app.HighScores(row, 1:index-1) app.Score app.HighScores(row, index:5)];
                app.HighScorers(row, 1:6) = [app.HighScorers(row, 1:index-1) app.PlayerName app.HighScorers(row, index:5)];
    
                % Update the file
                file = fopen(app.Filename, "w");
                for row = 1:3
                    fprintf(file, "%s\t", app.HighScorers(row, 1:6));
                    fprintf(file, "%s\n", "");
                end
                for row = 1:3
                    fprintf(file, "%d\t", app.HighScores(row, 1:6));
                    fprintf(file, "%s\n", "");
                end
                fclose(file);
            end
        end
        
        
    end

            app.HighScoreTitle = uilabel(app.HighScoresTab);
            app.HighScoreTitle.HorizontalAlignment = 'center';
            app.HighScoreTitle.FontName = 'Comic Sans MS';
            app.HighScoreTitle.FontSize = 70;
            app.HighScoreTitle.FontWeight = 'bold';
            app.HighScoreTitle.FontColor = [0.949 0.7686 0.0314];
            app.HighScoreTitle.Position = [699 844 422 120];
            app.HighScoreTitle.Text = 'High Scores';

            % Show the figure after all components are created
            app.UIFigure.Visible = 'on';
        end
    end

    % App creation and deletion
    methods (Access = public)

        % Construct app
        function app = Assignment

            % Create UIFigure and components
            createComponents(app)

            % Register the app with App Designer
            registerApp(app, app.UIFigure)

            % Execute the startup function
            runStartupFcn(app, @startupFcn)

            if nargout == 0
                clear app
            end
        end

        % Code that executes before app deletion
        function delete(app)

            % Delete UIFigure when app is deleted
            delete(app.UIFigure)
        end
    end
